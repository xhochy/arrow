if (ARROW_BUILD_TESTS)
  add_library(arrow_test_main
	test_main.cc)

  if (APPLE)
	  target_link_libraries(arrow_test_main
        gtest
        dl)
	  set_target_properties(arrow_test_main
        PROPERTIES LINK_FLAGS "-undefined dynamic_lookup")
  elseif(MSVC)
	  target_link_libraries(arrow_test_main
        gtest)
  else()
	  target_link_libraries(arrow_test_main
        gtest
        pthread
        dl
	    )
  endif()
endif()

      benchmark
      pthread
	  )
  endif()
endif()
